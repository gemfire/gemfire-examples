Distributed Types API 
DSet<String> set = factory.createDSet("demo-dset");
boolean added = set.add("Session-XYZ"); // Distributed uniqueness check
DList<String> list = factory.createDList("demo-dlist");
list.add("Event-1"); // Ordered cluster-wide list
DBlockingQueue<String> q = factory.createDBlockingQueue("demo-dbq", 100);
q.put("Task"); // Blocks if full
String t = q.take(); // Blocks until item is available
DCircularQueue<String> cq = factory.createDCircularQueue("demo-dcq", 3);
cq.add("Item-4"); // FIFO with fixed capacity (auto-eviction)
DAtomicLong al = factory.createDAtomicLong("demo-dal", 0L);
al.addAndGet(1L); // Atomic increment across all JVMs
DAtomicReference<String> ar = factory.createDAtomicReference("demo-dar", "IDLE");
ar.set("ACTIVE"); // Thread-safe cluster-wide ref
DCounter c = factory.createDCounter("demo-dcounter", 0);
c.increment(5); // Optimized high-throughput CRDT counter
DSemaphore sem = factory.createDSemaphore("demo-dsemaphore", 3);
sem.acquire(); // Shared cluster-wide permit management
DCountDownLatch dl = factory.createDCountDownLatch("demo-dcdl", 3);
dl.countDown(); // Decrement global signal
DSnowflake snowflake = DSnowflake.builder().build();
long id = snowflake.nextId(); // Time-sortable unique identifier
